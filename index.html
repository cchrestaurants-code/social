⇪ Google Sheets Sync

Push data to Sheets after every scan. Anyone on another computer can then click Load from Sheets to restore all monthly history into their browser.

📋 Setup (do this once — or redeploy if you had the old script): 1. Open your Google Sheet → Extensions → Apps Script
2. Replace ALL existing code with the script below → Save
3. Click Deploy → New Deployment (not Update) → Web App
4. Execute as: Me · Who has access: Anyone
5. Copy the new Web App URL → paste in the field above
6. ↑ Push saves your data to Sheets · ↓ Load restores it to any browser

Apps Script — paste this (supports both read and write):
function doPost(e){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var sheet=ss.getSheetByName('SocialData'); if(!sheet)sheet=ss.insertSheet('SocialData'); var payload=JSON.parse(e.postData.contents); if(sheet.getLastRow()===0){ sheet.appendRow(payload.headers); var hRange=sheet.getRange(1,1,1,payload.headers.length); hRange.setBackground('#1a1a1a'); hRange.setFontColor('#D4AF37'); hRange.setFontWeight('bold'); } payload.rows.forEach(function(row){sheet.appendRow(row);}); return ContentService .createTextOutput(JSON.stringify({status:'ok',rows:payload.rows.length})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e){ var ss=SpreadsheetApp.getActiveSpreadsheet(); var sheet=ss.getSheetByName('SocialData'); if(!sheet)return ContentService.createTextOutput(JSON.stringify({rows:[]})).setMimeType(ContentService.MimeType.JSON); var data=sheet.getDataRange().getValues(); var headers=data[0]; var rows=data.slice(1).filter(function(r){return r[0]!='';}); return ContentService .createTextOutput(JSON.stringify({headers:headers,rows:rows})) .setMimeType(ContentService.MimeType.JSON); }
